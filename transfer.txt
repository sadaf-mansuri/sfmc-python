import os
import shutil

# Define source and destination directories
original_dir = "path/to/original_dir"
processed_dir = "path/to/processed_dir"

# Ensure the destination directory exists
os.makedirs(processed_dir, exist_ok=True)

# Loop through all files in the original directory
for filename in os.listdir(original_dir):
    # Construct full file path
    source_file = os.path.join(original_dir, filename)
    
    # Check if it is a file (not a directory)
    if os.path.isfile(source_file):
        # Define the new file name (you can customize this)
        new_filename = f"processed_{filename}"
        destination_file = os.path.join(processed_dir, new_filename)
        
        # Copy the file
        shutil.copy2(source_file, destination_file)
        print(f"Copied: {filename} to {new_filename}")
